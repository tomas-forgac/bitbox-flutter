writeUInt64LE function

int writeUInt64LE (ByteData bytes, int offset, int value)

Implementation

int writeUInt64LE (ByteData bytes, int offset, int value) {
  isUint(value, 64);
  bytes.setInt32(offset, value & -1, Endian.little);
  bytes.setUint32(offset + 4, value ~/ 0x100000000, Endian.little);
  return offset + 8;
}
